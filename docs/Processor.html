def self.process(body, msg_id)
  xenapi = XenApi.new('server_path', nil, 'root', 'chnage_me')
  rabbit = Rabbit.new
  parsed = JSON.parse(body)
  payload = parsed['payload']
  puts ' [x] Task : ' + parsed['task']
  msg = {
    seq: parsed['id'],
    taskid: parsed['uuid'],
    timestamp: Time.now.to_s,
    payload:          case parsed['task']
      when 'get.vdi.all'
        xenapi.vdi_list('include')
      when 'get.vdi.iso'
        xenapi.vdi_list('only')
      when 'get.vdi.disk'
        xenapi.vdi_list('exclude')
      when 'get.vdi.my'
        xenapi.vdi_search_by_tag('userid:' + payload)
      when 'get.vdi.snapshot'
        xenapi.vdi_list_snapshot
      when 'get.vdi.tools'
        xenapi.vdi_list_tools
      when 'get.vdi.detail'
        xenapi.vdi_get_record(payload)
      when 'do.vdi.resize'
        Messages.error_unsupported
        # xenapi.vdi_resize(payload['vdi-ref'], payload['vdi-new-size'])
      when 'set.vdi.tag'
        xenapi.vdi_add_tag(payload['vm'], payload['tag'])
      when 'no.set.vm.tag'
        xenapi.vdi_rm_tag(payload['vm'], payload['tag'])
      when 'do.vdi.destroy'
        xenapi.vdi_destroy(payload)
      when 'get.vbd.all'
        xenapi.vbd_list
      when 'get.vbd.detail'
        xenapi.vbd_get_detail(payload)
      when 'do.vbd.create'
        xenapi.vbd_create(payload['vm-ref'], payload['vdi-ref'], payload['vm-slot'])
      else
        Messages.error_undefined
      end
  }
  xenapi.session_logout
  rabbit.publish(JSON.generate(msg), msg_id)
end
