def self.error_switch(error_string)
  case error_string
  when 'VM_BAD_POWER_STATE'
    error_vm_bad_power_state
  when 'OTHER_OPERATION_IN_PROGRESS'
    error_try_later
  when 'OPERATION_NOT_ALLOWED' || 'VM IS TEMPLATE'
    error_not_permitted
  when 'SR_FULL'
    error_disk_full
  else
    error_unknown(error_string)
  end
end
